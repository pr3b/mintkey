# dyte-camera-toggle

```html
<dyte-camera-toggle size="sm" class="dyte-el"></dyte-camera-toggle>
<dyte-camera-toggle size="lg" class="dyte-el"></dyte-camera-toggle>
<dyte-camera-toggle variant="horizontal" class="dyte-el"></dyte-camera-toggle>

<script>
  const elements = document.getElementsByClassName('dyte-el');
  for (const el of elements) {
    el.meeting = meeting;

```
